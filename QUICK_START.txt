🔧 STACK TECHNIQUE

⚠️ RÈGLES CRITIQUES
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

2. TOUJOURS exclure _id dans MongoDB : {"_id": 0}

4. PRÉFIXER routes backend par /api
5. UTILISER datetime.now(timezone.utc) pas utcnow()
